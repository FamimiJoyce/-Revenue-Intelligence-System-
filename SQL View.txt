 1.CREATE OR REPLACE VIEW pipeline_summary AS
SELECT 
    d.stage AS stage_name,
    COUNT(d.deal_id) AS total_deals,
    SUM(d.deal_value) AS total_value,
    AVG(EXTRACT(DAY FROM (d.close_date - d.created_at))) AS avg_deal_age
FROM deals d
GROUP BY d.stage
ORDER BY total_value DESC;


2. CREATE O'R REPLACE VIEW revenue_summary AS
SELECT 
    EXTRACT(MONTH FROM r.received_date) AS month,
    p.name AS product_name,
    s.name AS rep_name,
    SUM(r.amount) AS total_revenue
FROM revenue r
JOIN deals d ON r.deal_id = d.deal_id
JOIN products p ON d.product_id = p.product_id
JOIN sales_reps s ON d.rep_id = s.rep_id
GROUP BY month, product_name, rep_name
ORDER BY month, total_revenue DESC;

3. CREATE OR REPLACE VIEW top_customers AS
SELECT 
    c.customer_id,
    c.name,
    c.segment,
    COUNT(d.deal_id) AS deals_count,
    SUM(d.deal_value) AS total_deal_value
FROM customers c
JOIN deals d ON c.customer_id = d.customer_id
GROUP BY c.customer_id, c.name, c.segment
ORDER BY total_deal_value DESC
LIMIT 20;

 4.CREATE OR REPLACE VIEW churn_risk_report AS
SELECT 
    c.customer_id,
    c.name,
    cs.risk_level,
    cs.last_activity_date,
    COUNT(d.deal_id) AS active_deals
FROM customers c
JOIN churn_signals cs ON c.customer_id = cs.customer_id
LEFT JOIN deals d ON c.customer_id = d.customer_id
GROUP BY c.customer_id, c.name, cs.risk_level, cs.last_activity_date
ORDER BY cs.risk_level DESC, last_activity_date ASC;


 5.CREATE OR REPLACE VIEW forecast_vs_actual AS
SELECT
    f.rep_id,
    s.name AS rep_name,
    f.month,
    f.predicted_revenue,
    COALESCE(SUM(r.amount),0) AS actual_revenue
FROM forecasts f
JOIN sales_reps s ON f.rep_id = s.rep_id
LEFT JOIN deals d ON d.rep_id = f.rep_id
LEFT JOIN revenue r ON r.deal_id = d.deal_id AND EXTRACT(MONTH FROM r.received_date) = EXTRACT(MONTH FROM f.month)
GROUP BY f.rep_id, rep_name, f.month, f.predicted_revenue
ORDER BY f.month, rep_name;

 6.CREATE OR REPLACE VIEW opportunity_scores_view AS
SELECT 
    d.deal_id,
    c.name AS customer_name,
    s.name AS rep_name,
    o.score,
    o.next_best_action,
    d.stage,
    d.deal_value
FROM opportunity_scores o
JOIN deals d ON o.deal_id = d.deal_id
JOIN customers c ON d.customer_id = c.customer_id
JOIN sales_reps s ON d.rep_id = s.rep_id
ORDER BY score DESC;

 7.CREATE OR REPLACE VIEW deal_activity_summary AS
SELECT
    d.deal_id,
    c.name AS customer_name,
    s.name AS rep_name,
    d.stage,
    COUNT(a.activity_id) AS total_activities,
    MAX(a.activity_date) AS last_activity_date
FROM deals d
JOIN deal_activities a ON d.deal_id = a.deal_id
JOIN customers c ON d.customer_id = c.customer_id
JOIN sales_reps s ON d.rep_id = s.rep_id
GROUP BY d.deal_id, c.name, s.name, d.stage
ORDER BY last_activity_date DESC;

 8.CREATE OR REPLACE VIEW monthly_revenue_trend AS
SELECT
    DATE_TRUNC('month', r.received_date) AS month,
    SUM(r.amount) AS total_revenue,
    COUNT(DISTINCT r.deal_id) AS total_deals
FROM revenue r
GROUP BY month
ORDER BY month;


 9.CREATE OR REPLACE VIEW customer_segments AS
SELECT
    c.segment,
    COUNT(DISTINCT c.customer_id) AS total_customers,
    COUNT(DISTINCT d.deal_id) AS total_deals,
    SUM(d.deal_value) AS total_deal_value,
    SUM(COALESCE(r.amount,0)) AS total_revenue
FROM customers c
LEFT JOIN deals d ON c.customer_id = d.customer_id
LEFT JOIN revenue r ON d.deal_id = r.deal_id
GROUP BY c.segment
ORDER BY total_revenue DESC;

10. CREATE OR REPLACE VIEW rep_performance AS
SELECT
    s.rep_id,
    s.name AS rep_name,
    COUNT(DISTINCT d.deal_id) AS total_deals,
    SUM(d.deal_value) AS total_deal_value,
    SUM(COALESCE(r.amount,0)) AS total_revenue
FROM sales_reps s
LEFT JOIN deals d ON s.rep_id = d.rep_id
LEFT JOIN revenue r ON d.deal_id = r.deal_id
GROUP BY s.rep_id, s.name
ORDER BY total_revenue DESC;
